local RunService = game:GetService("RunService") local Players = game:GetService("Players") local MapFolder = workspace:WaitForChild("MapFolder") local Player = Players.LocalPlayer local Character = Player.Character or Player.CharacterAdded:Wait() local HumanoidRootPart = Character:WaitForChild("HumanoidRootPart") local MiniMapFrame = script.Parent:WaitForChild("MapFrame") local PlayerArrow = MiniMapFrame:WaitForChild("PlayerArrow") local MapCamera = Instance.new("Camera") MapCamera.CameraType = Enum.CameraType.Scriptable MapCamera.FieldOfView = 1 MiniMapFrame.CurrentCamera = MapCamera for _, object in ipairs(MapFolder:GetChildren()) do local clone = object:Clone() clone.Parent = MiniMapFrame end RunService.RenderStepped:Connect(function() local cameraPosition = HumanoidRootPart.Position + Vector3.new(0, 5000, 0) MapCamera.CFrame = CFrame.new(cameraPosition, HumanoidRootPart.Position) PlayerArrow.Rotation = -HumanoidRootPart.Orientation.Y - 90 end)
